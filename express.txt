    res.json({

use of post and message of the body in terminal using body parse using postman
const express = require('express');
const bodyparser=require('body-parser');
const app = express();
const port = 3500;

app.use(bodyparser.json());

app.post('/profile', (req, res) => {
  console.log(req.body);   // shows Postman message in terminal
  res.json("welcome to my website");
});

app.listen(port, () => {
  console.log(`server running on port ${port}`);
});
