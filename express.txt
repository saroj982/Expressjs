//use of function to calculate sum of n numbers use of postman 
const express=require('express');
const app=express();
const port=4000;
function calculateSum(n){
  let ans=0;
  for(let i=0; i<=n;i++){
    ans=ans+i;
  }
  return ans;
}
app.get('/',function(req,res){
  res.send("hello world");
})

app.post('/',(req,res)=>{
  const n=req.query.n;
  console.log(n);
  const ans=calculateSum(n);
  res.send(ans);
});

app.listen(port,()=>{
  console.log(`the server is running on port${port}`);
});
